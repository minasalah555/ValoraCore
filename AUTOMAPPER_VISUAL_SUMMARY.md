?  ?? USAGE EXAMPLE                                                    ?
????????????????????????????????????????????????????????????????????????
?                                                                      ?
?  // Inject AutoMapper                                               ?
?  private readonly IMapper _mapper;                                  ?
?                                                                      ?
?  public ReviewsController(IMapper mapper)                           ?
?  {                                                                   ?
?      _mapper = mapper;                                              ?
?  }                                                                   ?
?                                                                      ?
?  // GET: Entity ? DTO (Single Object)                               ?
?  var review = await _repository.GetByID(id);                        ?
?  var dto = _mapper.Map<ReviewDTO>(review);                          ?
?                                                                      ?
?  // GET: Entity ? DTO (Collection)                                  ?
?  var reviews = await _repository.GetAll();                          ?
?  var dtos = _mapper.Map<List<ReviewDTO>>(reviews);                  ?
?                                                                      ?
?  // POST: ViewModel ? Entity                                        ?
?  var review = _mapper.Map<Review>(createViewModel);                 ?
?  await _repository.Add(review);                                     ?
?                                                                      ?
?  // PUT: ViewModel ? Existing Entity                                ?
?  var review = await _repository.GetByID(id);                        ?
?  _mapper.Map(updateViewModel, review);  // Updates in-place         ?
?  await _repository.Update(review);                                  ?
